def sortings(data):
    li={}
    col=data.columns
    for i in col:
        index=0
        for j in data[i]:
            if j in li:
                continue
            li[j]=index
            index+=1
        for j in range(len(data[i])):
            for k in li:
                if data[i][j]==k:
                    data[i][j]=li[k]
        li.clear()
    return data

def get_dummis(data):
    """
    Must data include "label" column
    """
    label=data["label"]
    dic={}
    li=[]
    index=0
    for i in label:
        if i in dic:
            continue
        dic[i]=0
        li.append(0)
    for i in dic:
        dic[i]=li.copy() # 같은 주소 공유 방지
    for i in dic:
        dic[i][index]=1
        # print(dic[i],i)
        index+=1
    for i,v in enumerate(data["label"]):
        for j in dic:
            if v==j:# "fat"=="fat"
                data["label"][i]=dic[j]
    return data
